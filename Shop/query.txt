CREATE TABLE Category (
	ID INT NOT NULL,
	name VARCHAR(50),
	PRIMARY KEY (ID)
);
 
CREATE TABLE Item (
	ARTICLENUMBER INT NOT NULL,
	name VARCHAR(50),
	categoryID INT,
	amount INT(50),
	price DECIMAL(18, 2),
	PRIMARY KEY (ARTICLENUMBER)
);

CREATE TABLE Person (
	PERSONALNUMBER VARCHAR(50) NOT NULL,
	name VARCHAR(50),
	street VARCHAR(100),
	postnumber VARCHAR(100),
	PRIMARY KEY (PERSONALNUMBER)
);

CREATE TABLE Customer (
	PERSONALNUMBER VARCHAR(50) NOT NULL,
	username VARCHAR(50),
	password VARCHAR(50),
	PRIMARY KEY (PERSONALNUMBER)
);

CREATE TABLE Staff (
	PERSONALNUMBER VARCHAR(50) NOT NULL,
	username VARCHAR(50),
	password VARCHAR(50),
	PRIMARY KEY (PERSONALNUMBER)
);


CREATE TABLE Phone (
	PERSONALNUMBER VARCHAR(50) NOT NULL,
	phonenumber BIGINT,
	PRIMARY KEY (PERSONALNUMBER)
);

CREATE TABLE Rating (
	ARTICLENUMBER INT NOT NULL,
	personalnumber VARCHAR(50) NOT NULL,
	rate INT
);
