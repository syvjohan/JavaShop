CREATE TABLE Category (
	ID INT NOT NULL,
	name VARCHAR(50),
	PRIMARY KEY (ID)
);
 
CREATE TABLE Item (
	ID INT NOT NULL,
	name VARCHAR(50),
	categoryID INT,
	size INT,
	price DECIMAL(18, 2),
	PRIMARY KEY (ID)
);

CREATE TABLE Person (
	PERSONALNUMBER VARCHAR(50) NOT NULL,
	name VARCHAR(50),
	street VARCHAR(100),
	postnumber VARCHAR(100),
	postalAddress VARCHAR(100),
	PRIMARY KEY (PERSONALNUMBER)
);

CREATE TABLE Customer (
	PERSONALNUMBER VARCHAR(50) NOT NULL,
	username VARCHAR(50),
	password VARCHAR(50),
	PRIMARY KEY (PERSONALNUMBER)
);

CREATE TABLE Staff (
	PERSONALNUMBER VARCHAR(50) NOT NULL,
	username VARCHAR(50),
	password VARCHAR(50),
	PRIMARY KEY (PERSONALNUMBER)
);


CREATE TABLE Phone (
	PERSONALNUMBER VARCHAR(50) NOT NULL,
	phonenumber BIGINT,
	PRIMARY KEY (PERSONALNUMBER)
);

CREATE TABLE Rating (
	itemID INT NOT NULL,
	personalnumber VARCHAR(50) NOT NULL,
	rate INT
);
